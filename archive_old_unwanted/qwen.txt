class EvidenceProvenanceChain:
    def build_chain(self, claim, evidence_items):
        chain = {
            'claim': claim,
            'evidence_items': [],
            'support_strength': 0.0,
            'challenge_strength': 0.0,

        chain['support_strength'] = self.calculate_support(chain)
        chain['challenge_strength'] = self.calculate_challenge(chain)
        chain['independence_score'] = self.calculate_independence(chain)
        
        return chain

# Source Independence Graph Algorithm
class SourceIndependenceDetector:
    def detect_independence(self, sources):
        """
        Build citation graph and detect true independent sources
        """
        citation_graph = self.build_citation_graph(sources)
        independence_clusters = self.cluster_by_origin(citation_graph)
        
        true_independent_count = len(independence_clusters)
        apparent_count = len(sources)
        inflation_factor = apparent_count / true_independent_count
        
        return {
            'total_sources': apparent_count,
            'independent_sources': true_independent_count,
            'inflation_factor': inflation_factor,
            'independence_graph': citation_graph,
            'clusters': independence_clusters
        }
    
    def build_citation_graph(self, sources):
        """
        Extract citation/reference links from each source
        Check if Source B's claims appear verbatim in Source A
        Use publication timestamps for temporal direction
        """
        graph = {}
        for source in sources:
            graph[source.id] = {
                'cites': self.extract_citations(source),
                'cited_by': [],
                'timestamp': source.published_at,
                'content_hash': self.hash_content(source)
            }
        
        # Build reverse links
        for source_id, data in graph.items():
            for cited_id in data['cites']:
                if cited_id in graph:
                    graph[cited_id]['cited_by'].append(source_id)
        
        return graph
```

class AdversarialConfidenceModel:
    def assess_claim(self, claim, evidence, challenges):
        """
        Two-axis confidence: Support Strength + Challenge Strength
        NOT a single misleading number
        """
        support_strength = self.calculate_support_strength(
            independent_sources=evidence['independent_count'],
            source_quality=evidence['quality_scores'],
            source_diversity=evidence['class_diversity'],
            recency=evidence['freshness_scores'],
            specificity=evidence['specificity_scores']
        )
        
        challenge_strength = self.calculate_challenge_strength(
            contradictions=challenges['direct_contradictions'],
            missing_evidence=challenges['expected_but_missing'],
            source_quality=challenges['contradiction_quality'],
            logical_inconsistencies=challenges['logical_issues']
        )
        
        confidence_label = self.map_to_confidence_space(
            support_strength, challenge_strength
        )
        # Returns: settled, contested, uncertain, doubtful, insufficient
        
        return {
            'support_strength': support_strength,
            'challenge_strength': challenge_strength,
            'confidence_label': confidence_label,
            'recommendation': self.get_recommendation(confidence_label)
        }

# Cross-Evidence Deduction Engine
class CrossEvidenceDeductionEngine:
    def derive_deductions(self, evidence_graph):
        """
        Reason across complete evidence graph to derive new conclusions
        """
        deductions = []
        
        # 1. Temporal Pattern Detection
        temporal_deductions = self.detect_temporal_patterns(evidence_graph)
        deductions.extend(temporal_deductions)
        
        # 2. Cross-Lane Connections
        cross_lane_deductions = self.find_cross_lane_connections(evidence_graph)
        deductions.extend(cross_lane_deductions)
        
        # 3. Absence-as-Evidence Reasoning
        absence_deductions = self.reason_from_absence(evidence_graph)
        deductions.extend(absence_deductions)
        
        # 4. Convergence Signal Detection
        convergence_deductions = self.detect_convergence_signals(evidence_graph)
        deductions.extend(convergence_deductions)
        
        # 5. Implication Extraction
        implication_deductions = self.extract_implications(evidence_graph)
        deductions.extend(implication_deductions)
        
        return self.validate_deductions(deductions)
    
    def build_derivation_chain(self, deduction, evidence_items):
        """
        Every deduction MUST have a derivation chain
        This is mandatory for trust
        """
        chain = []
        for i, evidence in enumerate(evidence_items):
            chain.append({
                'step': i + 1,
                'evidence_id': evidence.id,
                'role_in_chain': evidence.role,
                'reasoning_step': self.generate_reasoning_text(evidence),
                'confidence_contribution': evidence.confidence_weight
            })
        
        return {
            'deduction': deduction.statement,
            'derivation_chain': chain,
            'deduction_type': deduction.type,
            'falsification_condition': deduction.falsification,
            'confidence': self.calculate_deduction_confidence(chain)
        }
```
